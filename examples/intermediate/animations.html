CSS Animations

This page demonstrates CSS animations for creating dynamic and engaging web experiences.

Basic Animation

/* Define keyframes */ @keyframes slideIn { from { transform: translateX(-100%); opacity: 0; } to { transform: translateX(0); opacity: 1; } } /* Apply animation */ .animated-element { animation: slideIn 2s ease-in-out; } /* Animation properties breakdown */ .detailed-animation { animation-name: slideIn; animation-duration: 2s; animation-timing-function: ease-in-out; animation-delay: 0s; animation-iteration-count: 1; animation-direction: normal; animation-fill-mode: forwards; }

Scale Animation

Animation Properties

/* Animation duration */ .fast-animation { animation-duration: 0.5s; } .slow-animation { animation-duration: 3s; } /* Animation timing functions */ .linear-animation { animation-timing-function: linear; } .ease-animation { animation-timing-function: ease; } .ease-in-animation { animation-timing-function: ease-in; } .ease-out-animation { animation-timing-function: ease-out; } .ease-in-out-animation { animation-timing-function: ease-in-out; } /* Animation iteration count */ .once-animation { animation-iteration-count: 1; } .infinite-animation { animation-iteration-count: infinite; } .multiple-animation { animation-iteration-count: 3; } /* Animation direction */ .normal-direction { animation-direction: normal; } .reverse-direction { animation-direction: reverse; } .alternate-direction { animation-direction: alternate; } .alternate-reverse-direction { animation-direction: alternate-reverse; } /* Animation fill mode */ .none-fill { animation-fill-mode: none; } .forwards-fill { animation-fill-mode: forwards; } .backwards-fill { animation-fill-mode: backwards; } .both-fill { animation-fill-mode: both; }

Animation Control

/* Play state */ .paused-animation { animation-play-state: paused; } .running-animation { animation-play-state: running; } /* Animation delay */ .delayed-animation { animation-delay: 2s; } /* Multiple animations */ .multiple-animations { animation: rotate 2s linear infinite, pulse 1s ease-in-out infinite; } /* Animation shorthand */ .animation-shorthand { animation: slideIn 2s ease-in-out 0s 1 normal forwards; }

Performance Considerations

/* Use transform and opacity for better performance */ @keyframes performant { 0%, 100% { transform: translateX(0); opacity: 1; } 50% { transform: translateX(100px); opacity: 0.5; } } /* Avoid animating these properties */ /* width, height, padding, margin, left, top */ /* Use will-change for complex animations */ .will-change-element { will-change: transform, opacity; } /* Remove will-change after animation */ .animation-complete { will-change: auto; } /* Use hardware acceleration */ .hardware-accelerated { transform: translateZ(0); backface-visibility: hidden; }

Best Practices

/* Keep animations short and purposeful */ .good-animation { animation-duration: 0.3s; } /* Use consistent timing across your site */ :root { --animation-fast: 0.15s ease; --animation-normal: 0.3s ease; --animation-slow: 0.5s ease; } /* Consider accessibility */ @media (prefers-reduced-motion: reduce) { * { animation: none !important; transition: none !important; } } /* Add meaningful animations */ .meaningful-animation { animation: slideIn 0.3s ease-out; /* Indicates entrance */ } /* Test on mobile devices */ .mobile-friendly { animation: pulse 2s ease-in-out infinite; /* Works well on touch */ }
