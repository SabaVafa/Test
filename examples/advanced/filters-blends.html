Practical Examples

/* Hover effects with filters */ .image-hover img { transition: filter 0.3s ease; } .image-hover:hover img { filter: brightness(1.2) contrast(1.1) saturate(1.2); } /* Loading state with blur */ .loading { filter: blur(2px); transition: filter 0.3s ease; } .loading.loaded { filter: none; } /* Disabled state with grayscale */ .disabled { filter: grayscale(100%) opacity(0.5); } /* Focus state with brightness */ .card:focus { filter: brightness(1.1); }

Performance Considerations

/* Filters are GPU-accelerated in modern browsers */ /* But can be expensive for complex operations */ /* Use transform instead of filters when possible */ .better-performance { transform: scale(1.1); /* Better than filter brightness */ } /* Avoid animating filters on large elements */ .expensive-animation { filter: blur(5px); transition: filter 0.3s ease; /* Can be slow */ } /* Use will-change for complex filter animations */ .complex-filter-animation { will-change: filter; } /* Remove will-change after animation */ .animation-complete { will-change: auto; }

Browser Support

/* Filter support */ /* Chrome 18+, Firefox 35+, Safari 6+, Edge 12+, IE 13+ */ /* Mix blend mode support */ /* Chrome 41+, Firefox 32+, Safari 8+, Edge 12+, IE 13+ */ /* Background blend mode support */ /* Chrome 41+, Firefox 30+, Safari 8+, Edge 12+, IE 13+ */ /* Backdrop filter support */ /* Chrome 76+, Firefox 103+, Safari 14+, Edge 79+, IE 13+ */ /* Prefixes for older browsers */ .element { -webkit-filter: blur(5px); filter: blur(5px); } /* Feature detection */ @supports (filter: blur(2px)) { /* Filters supported */ } @supports (backdrop-filter: blur(2px)) { /* Backdrop filter supported */ }

Best Practices

/* Use filters for visual effects, not layout */ /* Use transform for positioning and scaling */ /* Provide fallbacks for older browsers */ .fallback-filter { background: rgba(0,0,0,0.5); /* Fallback */ } @supports (backdrop-filter: blur(5px)) { .fallback-filter { background: rgba(255,255,255,0.1); backdrop-filter: blur(5px); } } /* Use filters sparingly for performance */ .performant { filter: brightness(1.1); /* Lightweight */ } /* Test on mobile devices */ /* Some filters can be slow on mobile */ /* Consider accessibility */ @media (prefers-reduced-motion: reduce) { .animated-filter { filter: none; } }
